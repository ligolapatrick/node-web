  <script>
    function fetchMyPosts() {
      fetch('/my-posts')
        .then(res => res.json())
        .then(data => {
          const container = document.getElementById('postList');
          container.innerHTML = '';
          data.forEach(post => {
            const card = document.createElement('div');
            card.className = 'post-card';
            card.innerHTML = `
              <h3>${post.title}</h3>
              <p>${post.content}</p>
              <small>Posted on ${new Date(post.postedAt).toLocaleString()}</small>
              ${post.media ? (post.media.endsWith('.mp4') ?
                `<video src="${post.media}" controls></video>` :
                `<img src="${post.media}" alt="Media" />`) : ''}
              <button class="edit-btn" onclick="editPost(${post.id})">‚úèÔ∏è Edit</button>
              <button class="delete-btn" onclick="deletePost(${post.id})">üóëÔ∏è Delete</button>
            `;
            container.appendChild(card);
          });
        });
    }

    function editPost(id) {
      window.location.href = `/createnews.html?id=${id}`;
    }

    function deletePost(id) {
      if (!confirm('Are you sure you want to delete this post?')) return;

      fetch('/delete-news', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ id })
      }).then(res => {
        if (res.ok) {
          fetchMyPosts();
        } else {
          alert('Failed to delete post');
        }
      });
    }

    fetchMyPosts();
  </script>
